# Financial RAG Chatbot Dependencies
# Python 3.12+ required
# Recommended: Use virtual environment (python -m venv myenv)

# =============================================================================
# Core Web Framework
# =============================================================================
streamlit==1.31.0  # Web interface for the chatbot

# =============================================================================
# Machine Learning & NLP Core
# =============================================================================
torch==2.2.0  # Deep learning framework
torchvision==0.17.0  # Computer vision library for torch
transformers==4.35.0  # Hugging Face transformers for T5 and other models
tokenizers==0.14.1  # Fast tokenizers (pre-built wheel for Python 3.12)
sentence-transformers==2.2.2  # For semantic embeddings (all-MiniLM-L6-v2)
huggingface-hub==0.17.3  # Model hub integration

# =============================================================================
# Vector Search & Retrieval
# =============================================================================
faiss-cpu==1.8.0  # Facebook AI Similarity Search for vector storage
rank-bm25==0.2.2  # BM25 algorithm for keyword-based retrieval

# =============================================================================
# Scientific Computing
# =============================================================================
numpy==1.26.4  # Numerical computing (pinned < 2.0 for streamlit compatibility)
scipy==1.11.4  # Scientific computing utilities
scikit-learn==1.3.0  # Machine learning utilities (Python 3.12 compatible)

# =============================================================================
# Data Processing & Analysis
# =============================================================================
pandas==2.3.2  # Data manipulation and analysis

# =============================================================================
# Document Processing
# =============================================================================
beautifulsoup4==4.12.2  # HTML parsing for web documents
PyPDF2==3.0.1  # PDF reading and manipulation
pdfplumber==0.10.2  # Advanced PDF extraction with table support
python-docx==1.0.1  # Microsoft Word document processing

# =============================================================================
# Serialization & Configuration
# =============================================================================
protobuf==3.20.3  # Protocol buffers for data serialization
pyyaml==6.0.1  # YAML parser for configuration files

# =============================================================================
# Installation Notes:
# =============================================================================
# 1. Python 3.12+ is required for all packages
# 2. Create virtual environment: python -m venv myenv
# 3. Activate: source myenv/bin/activate (Linux/Mac) or myenv\Scripts\activate (Windows)
# 4. Install: pip install -r requirements.txt
#
# Known Issues:
# - numpy must be < 2.0 for streamlit 1.31.0 compatibility
# - tokenizers 0.14.1 has pre-built wheels for Python 3.12 (no Rust compiler needed)
# - scikit-learn may take time to build from source (~10-15 minutes)
#
# For GPU support:
# - Replace torch==2.2.0 with torch==2.2.0+cu121 (for CUDA 12.1)
# - Replace faiss-cpu with faiss-gpu
# =============================================================================
